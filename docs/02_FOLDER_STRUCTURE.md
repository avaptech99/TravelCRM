## Backend: `travel-crm-backend/`

```
travel-crm-backend/
├── package.json                # Backend dependencies & scripts
├── package-lock.json           # Dependency lock
├── tsconfig.json               # TypeScript configuration
├── .env                        # ⚠️ PRIVATE — Environment variables (DO NOT COMMIT)
├── .env.example                # Template for .env file
├── dist/                       # Compiled JS output (generated by `npm run build`)
│
└── src/
    ├── server.ts               # 🚀 ENTRY POINT — Express app setup, routes, CORS, auto-seed, listen
    ├── seed.ts                 # Database seeder script (creates Admin + Agent demo accounts)
    ├── socket.ts               # Socket.io initialization (available for future real-time features)
    ├── importData.ts           # Data import utility (for migration from old systems)
    ├── test-mongo.js           # Quick MongoDB connection test script
    │
    ├── config/
    │   └── db.ts               # MongoDB connection setup (Mongoose) with pool config
    │
    ├── middleware/
    │   └── auth.ts             # JWT auth middleware (`protect`) + admin guard (`adminGuard`)
    │
    ├── models/                 # 📦 Mongoose schemas & models
    │   ├── Booking.ts          # Core booking entity (trip data, status, segments, amounts)
    │   ├── PrimaryContact.ts   # Contact person info (name, phone, email, requirements, interest)
    │   ├── Comment.ts          # Comments/remarks on bookings
    │   ├── Passenger.ts        # Passenger/traveler details for finalized bookings
    │   ├── Traveler.ts         # Legacy traveler model (similar to Passenger, kept for migration)
    │   ├── Payment.ts          # Payment records per booking
    │   ├── User.ts             # User accounts (Admin, Agent, Marketer)
    │   ├── Notification.ts     # In-app notifications (assignment alerts, status changes)
    │   ├── Activity.ts         # Activity/audit log entries per booking
    │   └── Counter.ts          # Auto-increment counter for sequential booking codes (TW0001, TW0002...)
    │
    ├── controllers/            # 🧠 Business logic handlers
    │   ├── authController.ts        # Login logic with performance logging + bcrypt hash upgrade
    │   ├── bookingController.ts     # 📌 LARGEST FILE (~1100 lines) — All booking CRUD + passengers + payments + calendar + activity
    │   ├── userController.ts        # User CRUD + password change + status tracking + unassign bookings
    │   ├── analyticsController.ts   # Aggregation queries for reports (bookings, payments, revenue, agents)
    │   ├── syncController.ts        # Single consolidated endpoint for dashboard data
    │   ├── notificationController.ts# Notification fetch + mark-as-read
    │   └── externalController.ts    # WordPress lead intake (raw field parsing, multi-city handling)
    │
    ├── routes/                 # 🛣️ Express route definitions
    │   ├── authRoutes.ts       # POST /login + GET /seed (temporary)
    │   ├── bookingRoutes.ts    # All /api/bookings/* routes (most complex)
    │   ├── userRoutes.ts       # All /api/users/* routes
    │   ├── analyticsRoutes.ts  # GET /api/analytics/* (admin-only)
    │   ├── syncRoutes.ts       # GET /api/sync (dashboard polling)
    │   ├── notificationRoutes.ts  # GET/PUT /api/notifications/*
    │   └── externalRoutes.ts   # POST /api/external/lead (WordPress intake)
    │
    ├── utils/                  # 🔧 Helper utilities
    │   ├── cache.ts            # In-memory TTL cache (Map-based, no Redis)
    │   ├── extractTravelInfo.ts# NLP extraction (chrono-node + compromise) with cache
    │   ├── jwt.ts              # JWT token generation & verification
    │   ├── password.ts         # bcrypt hash, compare, and round-upgrade logic
    │   ├── activityLogger.ts   # Writes activity log entries to the Activity model
    │   └── keepWarm.ts         # Self-ping every 10 minutes to prevent Render free-tier sleep
    │
    ├── types/
    │   └── index.ts            # Zod validation schemas for all API inputs
    │
    └── scripts/
        └── migrateData.ts      # Data migration script (e.g., moving data from old system)
```
